- Is there a seasonal pattern to shark attacks in different regions?

# Convert the date to a datetime object and extract the month
df['Date'] = pd.to_datetime(df['Date'])
df['Month'] = df['Date'].dt.month

# Group the data by region and month, and count the number of attacks
seasonal_pattern = df.groupby(['Region', 'Month'])['Attack'].count().reset_index()

# Plot the number of attacks for each month in each region
for region in seasonal_pattern['Region'].unique():
    plt.plot(seasonal_pattern[seasonal_pattern['Region'] == region]['Month'],
             seasonal_pattern[seasonal_pattern['Region'] == region]['Attack'])
    plt.title(f'Seasonal Pattern of Shark Attacks in {region}')
    plt.xlabel('Month')
    plt.ylabel('Number of Attacks')
    plt.show()
